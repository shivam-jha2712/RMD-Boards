                <script>
                  document.querySelectorAll('.field--request-type select').forEach(requestType => {
                    requestType.addEventListener('change', e => {
                      document.querySelector('.form-main').classList.toggle('unselected', !requestType.value);
                      if (requestType.value) {
                        document.querySelectorAll('[data-forms]').forEach(formElement => {
                          const relatedForms = formElement.dataset.forms,
                            regex = new RegExp(requestType.value)
                          formElement.classList.toggle('hidden', !regex.test(relatedForms));
                        });
                      }
                    });
                  });
                  document.querySelectorAll('form.form').forEach(form => form.addEventListener('submit', event => {
                    const action = form.getAttribute('action'),
                      isMock = /assets\/json/.test(action),
                      options = {};

                    form.classList.add('form--busy');

                    if (isMock) {
                      options.method = 'GET';
                    } else {
                      options.method = 'POST';
                      options.headers = {
                        'Accept': 'application/json',
                        'Content-Type': 'application/json'
                      };
                      options.body = {};

                      // Add visible fields
                      document.querySelectorAll('form.form .field:not(.hidden)').forEach(field => {
                        const input = field.querySelector('input, select, textarea');
                        let value = '';

                        if (input) {
                          if (input.type == 'checkbox') {
                            value = input.checked;
                          } else if (field.querySelector('.select-wrap--phone')) {
                            const code = field.querySelector('select').value,
                              phone = field.querySelector('input').value;
                            if (phone) {
                              value = `+${code} ${phone}`;
                            }
                          } else {
                            value = input.value;
                          }

                          if (field.dataset.fieldId) {
                            options.body[field.dataset.fieldId] = value;
                          } else {
                            options.body[input.name] = value;
                          }
                        }
                      });

                      // Add hidden (implicit) fields
                      document.querySelectorAll('form.form [type="hidden"]').forEach(input => {
                        options.body[input.name] = input.value;
                      });

                      options.body = JSON.stringify(options.body);
                    }

                    setTimeout(() => {
                      fetch(action, options)
                        .then(response => response.json())
                        .then(response => {
                          form.classList.remove('form--busy');
                          if (response.action) {
                            form.setAttribute('action', response.action);
                          }
                          if (response.status == 'ok') {
                            form.parentElement.classList.add('hidden');
                            document.querySelectorAll('.form-success')[0].classList.add("visible");
                            document.body.scrollIntoView({ behavior: 'smooth', block: 'start' });
                          } else {
                            let hasErrors = false;
                            let errors = response.fields;
                            let generalError = false;
                            let customError = false;
                            let customErrorContent = '';
                            for (var name in errors) {
                              if (errors.hasOwnProperty(name)) {
                                if (name === 'raw') {
                                  console.log(errors[name]);
                                } else if (name === 'zoho') {
                                  generalError = true;
                                } else if (name === 'custom') {
                                  customError = true;
                                  customErrorContent = errors[name];
                                } else {
                                  form.querySelectorAll(`[name="${name}"]`).forEach(input => {
                                    let wrapper = input.closest('.field:not(.hidden)');
                                    if (wrapper) {
                                      wrapper.classList.add('has-error');
                                      hasErrors = true;
                                    }
                                  });
                                }
                              }
                            }
                            form.querySelector('.row--general-error[data-error="general"]').classList.toggle('expanded', generalError);
                            form.querySelector('.row--general-error[data-error="custom"] p.error').innerHTML = customErrorContent;
                            form.querySelector('.row--general-error[data-error="custom"]').classList.toggle('expanded', customError);
                            form.querySelector('.row--error').classList.toggle('expanded', hasErrors);
                          }
                        })
                        .catch(() => {
                          form.classList.remove('form--busy');
                        })
                    }, isMock ? 1500 : 0);

                    event.preventDefault();
                  }));

                  document.querySelectorAll('.select-wrap select').forEach(select => {
                    select.addEventListener('change', () => {
                      const selectedOption = select.querySelector(':checked');
                      if (!selectedOption) { return; }
                      const placeholder = select.closest('.select-wrap').querySelector('.placeholder');
                      placeholder.innerText = selectedOption.dataset.placeholder || selectedOption.innerText;
                      placeholder.classList.toggle('empty', !selectedOption.value);
                      placeholder.closest('.field').classList.toggle('empty', !selectedOption.value);
                    });
                  });

                  document.querySelectorAll('[data-type="phone-code"]').forEach(phoneCode => {

                      if (phoneCode.activeElement) {
                        input.focus();
                      }
                    });
                  });

                  document.querySelectorAll('[data-type="country"]').forEach(country => {
                    country.addEventListener('change', () => {
                      document.querySelectorAll(`[data-country="${country.value}"]`).forEach(targetCode => {
                        targetCode.closest('select').value = targetCode.value;
                        targetCode.closest('select').dispatchEvent(new CustomEvent('change'));
                      });
                    });
                  });

                  document.querySelectorAll('form.form input, form.form textarea, form.form select').forEach(input => {
                    input.addEventListener('focus', () => {
                      const fieldWithError = input.closest('.has-error');
                      if (fieldWithError) {
                        fieldWithError.classList.remove('has-error');

                        // Nested errors?
                        const nestedError = fieldWithError.closest('.has-error');
                        if (nestedError) {
                          nestedError.classList.remove('has-error');
                        }
                      }
                      const remainingErrors = input.closest('.form').querySelectorAll('.field.has-error').length;
                      input.closest('.form').querySelectorAll('.row--error').forEach(rowError => rowError.classList.toggle('expanded', remainingErrors > 0));
                    });
                  });

                  document.querySelectorAll('form.form input:not([type="checkbox"]), form.form textarea, form.form select').forEach(input => {
                    const fieldclass = input.closest('.field')

                    input.addEventListener('focus', () => {
                      fieldclass.classList.add('active');
                    });

                    input.addEventListener('change', () => {
                      if (input.activeElement) {
                        fieldclass.classList.add('active');
                      }
                    });

                    input.addEventListener('blur', () => {
                      if (input.value.length == 0) {
                        fieldclass.classList.remove('active');
                      }
                    });

                  });

                  // Data action links
                  if (localStorage.contactFormData) {
                    const data = JSON.parse(localStorage.contactFormData);
                    Object.keys(data).forEach(key => {
                      let targets;

                      targets = document.querySelectorAll(`select[name="${key}"], select[data-type="${key}"]`);

                      targets.forEach(target => {
                        target.value = data[key];
                        target.dispatchEvent(new CustomEvent('change'));
                        target.closest('.field').classList.add('active');
                      });
                    });
                    localStorage.removeItem('contactFormData');
                  }
